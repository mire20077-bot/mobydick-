function toggleIcon(el, activeClass) {
  const classes = ['active','active-blue','active-yellow'];
  const isOn = el.classList.contains(activeClass);
  classes.forEach(c => el.classList.remove(c));
  if (!isOn) el.classList.add(activeClass);
}
